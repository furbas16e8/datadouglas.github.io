import polars as pl
# Lazy evaluation for memory efficiency
df = pl.scan_csv("large_dataset.csv")
    .filter(pl.col("status") == "active")
    .groupby("region")
    .agg([
        pl.col("sales").sum().alias("total")
    ])
    .collect()
